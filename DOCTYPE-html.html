<script>
// Danh sách từ HSK1 mẫu
const words = [
{ han: "你", pinyin: "nǐ", viet: "Bạn", meaning: "Bạn" },
{ han: "好", pinyin: "hǎo", viet: "Tốt", meaning: "Tốt" },
{ han: "人", pinyin: "rén", viet: "Người", meaning: "Người" },
{ han: "山", pinyin: "shān", viet: "Núi", meaning: "Núi" },
{ han: "水", pinyin: "shuǐ", viet: "Nước", meaning: "Nước" },
];
function showTab(n) {
document.querySelectorAll('.tab').forEach(tab => tab.classList.add('hidden'));
document.getElementById('tab' + n).classList.remove('hidden');
}
function startLearning() {
showTab(1);
const container = document.getElementById('lessonList');
container.innerHTML = '';
words.forEach(word => {
const div = document.createElement('div');
div.className = "bg-white rounded-3xl p-6 shadow-xl card text-center";
div.innerHTML = `
<div class="text-7xl mb-4">${word.han}</div>
<div class="text-2xl font-bold text-red-600">${word.pinyin}</div>
<div class="text-lg">${word.viet}</div>
<button onclick="this.parentElement.style.backgroundColor='#d1fae5'; alert('🐼
Đúng rồi! ${word.meaning}')"
class="mt-4 bg-blue-500 text-white px-6 py-3 rounded-2xl">
Học & Kiểm tra
</button>
`;
container.appendChild(div);
});
}
function playGame() {
showTab(2);
}
// Khởi động
showTab(0);
</script>
